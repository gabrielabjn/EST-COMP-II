meu_gerador <- function(n){
     
  u1 <- numeric(n)
  u2 <- numeric(n)
  s <- numeric(n)
  z <- numeric(n)
  x1 <- numeric(n)
  x2 <- numeric(n)
  
  for (i in 1:n) {
    s[i] <- 0
    
    while (s[i] >= 1 || s[i] == 0) {
      u1[i] <- runif(1, min = -1, max = 1)
      u2[i] <- runif(1, min = -1, max = 1)
      s[i] <- u1[i]**2 + u2[i]**2
    }
    
    z[i] <- sqrt(-2*log(s[i])/s[i])
    x1[i] <- z[i]*u1[i]
    x2[i] <- z[i]*u2[i]
  }
  
  return(list(x1 = x1, x2 = x2))
}

Questao 04 - Geracao de VAs Beta(1,2)


# a) Use o algoritmo de aceitação/rejeição baseado na distribuição uniforme.

rbeta_aceitacao_rejeicao <- function(n, a, b) {
  # Define a densidade da distribuição Beta(a,b)
  f_beta <- function(x) {
    if (x < 0 || x > 1) {
      return(0)
    }
    return(x^(a-1) * (1-x)^(b-1) / beta(a, b))
  }
  
  # Define a constante de normalização M
  M <- (a+b)^(-a-b+1) * max(1, (a-1) / (b-1))
  
  # Gerando amostras
  i <- 1
  x <- numeric(n)
  while (i <= n) {
    # Gerando uma amostra de x a partir de uma distribuição uniforme
    x_aleatorio <- runif(1)
    
    # Gerando uma amostra de y a partir de uma distribuição uniforme
    y_aleatorio <- runif(1, 0, M * f_beta(x_aleatorio))
    
    # Verificando se a amostra de y é menor que a densidade da distribuição Beta(a,b)
    if (y_aleatorio <= f_beta(x_aleatorio)) {
      x[i] <- x_aleatorio
      i <- i + 1
    }
  }
  
  # Retornando as amostras geradas
  return(x)
}

# b) Use o algoritmo de aceitação/rejeição baseado em beta gerada de acordo com a transformação apresentada no exemplo 2.2 de Robert e Casella (2010), pág. 69 (do pdf).


rbeta_transformacao <- function(n, a, b) {
  # Define a função f(x) e a função de envelope g(x)
  f <- function(x) x^(a-1) * (1-x)^(b-1) / beta(a,b)
  g <- function(x) 1
  
  # Define a constante de normalização M
  M <- max(f(seq(0,1,0.01)) / g(seq(0,1,0.01)))
  
  # Gerando amostras
  i <- 1
  x <- numeric(n)
  while (i <= n) {
    # Gerando uma amostra de y a partir de uma distribuição Beta(a,b)
    y <- rbeta(1, a, b)
    
    # Gerando uma amostra de u a partir de uma distribuição uniforme
    u <- runif(1)
    
    # Verificando se a amostra de u é menor ou igual a f(y)/(M*g(y))
    if (u <= f(y) / (M * g(y))) {
      x[i] <- y
      i <- i + 1
    }
  }
  
  # Retornando as amostras geradas
  return(x)
}

# c) Use o algoritmo de aceitação/rejeição baseado na distribuição normal truncada.

rnorm_truncada <- function(n, mu, sigma, a, b) {
  # Gerando amostras
  i <- 1
  x <- numeric(n)
  while (i <= n) {
    # Gerando uma amostra de y a partir de uma distribuição normal
    y <- mu + sigma * rnorm(1)
    
    # Verificando se a amostra de y está entre a e b
    if (y >= a && y <= b) {
      x[i] <- y
      i <- i + 1
    }
  }
  
  # Retornando as amostras geradas
  return(x)
}

# Seja X uma variável aleatória exponencial com média 1. Crie um código com um
# algoritmo eficiente para simular uma variável aleatória cuja distribuição é a
# distribuição condicional de X, dado que X < 0,05, ou seja, cuja função de
# densidade de probabilidade é:

fdp <- function(x){
  
  if (x > 0 & x < 0.05)
    return(exp(-x)/1-exp(-0.05))
  
  else
    return(0)
  
}


fda <- function(x){
  
  if (x <= 0)
    return(0)
  
  if(x > 0 & x < 0.05)
    return(exp(-x)-exp(-1/20))
  
  else
    return(1)
  
}
  
q <- function(p) { 
  ifelse(p <= 0, 0, ifelse(p > 1, 0.05, -log(1 - p*(1 - exp(-1/20)))))
}
  

gerador <- function(n = 1000){
  
  u = runif(n)
  x = q(u)
  
  return(x)
  
}
